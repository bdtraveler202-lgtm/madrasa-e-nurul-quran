// ====================================== // PUBLIC NOTICE BOARD // assets/notices.js // ====================================== let allNotices = []; // Load All Notices async function loadNotices() { const noticeBoard = document.getElementById("noticeBoard"); noticeBoard.innerHTML = `

নোটিশ লোড হচ্ছে...

`; const { data, error } = await window.supabaseClient .from("notices") .select("*") .order("pinned", { ascending: false }) .order("important", { ascending: false }) .order("created_at", { ascending: false }); if (error) { console.error(error); noticeBoard.innerHTML = `
❌ নোটিশ লোড করা যায়নি
`; return; } allNotices = data || []; renderNotices(allNotices); } // ====================================== // Render Notices // ====================================== function renderNotices(notices) { const noticeBoard = document.getElementById("noticeBoard"); noticeBoard.innerHTML = ""; if (notices.length === 0) { noticeBoard.innerHTML = `
কোনো নোটিশ পাওয়া যায়নি।
`; return; } notices.forEach(notice => { noticeBoard.innerHTML += `
${notice.image_url ? `` : ""}
${notice.pinned ? ` 📌 Pinned ` : ""} ${notice.important ? ` ⭐ Important ` : ""}
${notice.title}

${new Date(notice.created_at).toLocaleDateString("en-GB")}

${notice.description.length > 180 ? notice.description.substring(0,180) + "..." : notice.description}

${notice.pdf_url ? ` PDF ` : ""} Read More
`; }); } // ====================================== // Search // ====================================== const searchBox = document.getElementById("searchNotice"); if (searchBox) { searchBox.addEventListener("keyup", function () { const keyword = this.value.toLowerCase(); const filtered = allNotices.filter(notice => notice.title.toLowerCase().includes(keyword) || notice.description.toLowerCase().includes(keyword) ); renderNotices(filtered); }); } // ====================================== // Start // ====================================== loadNotices();
